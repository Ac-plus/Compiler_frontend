using namespace std;

set<string> keyword = {"int", "void", "return", "const"};
set<char> boundary = {'(', ')', '{', '}', ';', ','};
set<char> operation = {'+', '-', '*', '/', '%', '=', '>', '<'};
set<char> operationBeginChar = {'=', '<', '>', '!', '&', '|'};
set<string> operationOf2Char = {"==", "<=", ">=", "!=", "&&", "||"};
map<std::string, int> tokenCodeMap = {
    {"int", TokenCode::KW_INT},
    {"void", TokenCode::KW_VOID},
    {"return", TokenCode::KW_RETURN},
    {"const", TokenCode::KW_CONST},
    {"main", TokenCode::KW_MAIN},
    {"+", TokenCode::OP_PLUS},
    {"-", TokenCode::OP_MINUS},
    {"*", TokenCode::OP_MULTIPLY},
    {"/", TokenCode::OP_DIVIDE},
    {"%", TokenCode::OP_PERCENT},
    {"=", TokenCode::OP_ASSIGN},
    {">", TokenCode::OP_GT},
    {"<", TokenCode::OP_LT},
    {"==", TokenCode::OP_EQUAL},
    {"<=", TokenCode::OP_LEQ},
    {">=", TokenCode::OP_GEQ},
    {"!=", TokenCode::OP_NEQ},
    {"&&", TokenCode::OP_AND},
    {"||", TokenCode::OP_OR},
    {"(", TokenCode::SE_LPARENT},
    {")", TokenCode::SE_RPARENT},
    {"{", TokenCode::SE_LBRACES},
    {"}", TokenCode::SE_RBARCES},
    {";", TokenCode::SE_SEMICOLON},
    {",", TokenCode::SE_COMMA}
};

SymbolTable symbolTable = SymbolTable();
FSM NFA = createNFA();
FSM DFA = NFAtoDFA(NFA);
FSM miniDFA = minimizeDFA(DFA);
int lineNum = 0;

//bool lastWord = false;
//bool neg = false;

// 使用自动机分析token
void analyseToken(std::string token) {
    if(token.empty()) {
        return;
    }

    // 先判断是否为关键字
    std::string lowerToken = toLower(token);
    auto keywordIter = keyword.find(lowerToken);
    if(isAllLetter(lowerToken) && keywordIter != keyword.end()) {
        printToken(token, tokenCodeMap[lowerToken], lineNum);
        return;
    }

    // 用最小化DFA识别token得到TokenCode
    int tokenCode = identity(miniDFA, token);
    //if (neg) {
    //    if(tokenCode == TokenCode::INT) token = '-' + token;
    //    neg = false;
    //}
    printToken(token, tokenCode, lineNum);
    //lastWord = true;

    // 如果是标识符则添加至符号表中
    if(tokenCode == TokenCode::IDN) {
        symbolTable.addSymbol(token);
    }
}

void lexicalAnalysis(const std::string& fileName) {
    std::ofstream out(lexicalTxtPath);   // 打开输出文件并清空其内容
    out << std::flush;

    std::ifstream file(fileName);
    if (!file.is_open()) {
        std::cout << "Error: " << fileName << " could not open!" << std::endl;
        return;
    }

    char c;
    std::string token = "";
    while (file.get(c)) {
        cout << "READ CHAR : " << c << endl;
        if (std::isspace(c)) {  // 如果是空格、制表符或者换行符
            cout << "READ TOKEN : " << token << endl;
            analyseToken(token);
            token = "";
            if (c == '\n') {
                ++lineNum;
            }
        } else if (boundary.count(c)) { // 如果是界符
            cout << "READ TOKEN : " << token << endl;
            analyseToken(token);
            token = std::string(1, c);
            printToken(token, tokenCodeMap[token], lineNum);
            token = "";
        } else if (operation.count(c) || operationBeginChar.count(c)) { // 如果是运算符
            cout << "READ TOKEN : " << token << endl;
            analyseToken(token);
            if (operationBeginChar.count(c)) {  // 如果发现了可能是两个字符组成的运算符
                char nextChar = file.get();
                std::string tryOp = std::string(1, c) + nextChar;
                if (operationOf2Char.count(tryOp)) {  // 如果加上下一个字符是合法运算符
                    token = tryOp;
                    printToken(token, tokenCodeMap[token], lineNum);
                } else if (operation.count(c)) {  // 如果只是一个字符的运算符
                    token = std::string(1, c);
                    printToken(token, tokenCodeMap[token], lineNum);
                    file.putback(nextChar);
                } else {  // 如果是不合法的字符组合
                    token = std::string(1, c);
                    printToken(token, TokenCode::UNDIFNIE, lineNum);
                    file.putback(nextChar);
                }
            } else {  // 如果只是一个字符的运算符
                token = std::string(1, c);
                printToken(token, tokenCodeMap[token], lineNum);
            }
            token = "";
        } else {  // 如果既不是界符也不是运算符
            cout << "READ TOKEN : " << token << endl;
            token += c;
        }
    }
    if (!token.empty()) {  // 将最后一个 token 也进行处理
        analyseToken(token);
    }
    symbolTable.printTable();
}
